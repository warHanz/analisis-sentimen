# Perintah-perintah ini akan dieksekusi di terminal Streamlit Cloud sebelum aplikasi Anda berjalan.

# Langkah 1: Pastikan NLTK terinstal agar kita bisa menggunakan modulnya untuk download.
# Meskipun ada di requirements.txt, menjalankannya di sini memastikan NLTK ada saat perintah Python dijalankan.
pip install nltk

# Langkah 2: Unduh data NLTK yang diperlukan ke direktori yang umum dan dapat diakses oleh NLTK.
# /usr/share/nltk_data adalah lokasi standar di sistem Linux.
python -c "import nltk; nltk.download('punkt', download_dir='/usr/share/nltk_data')"
python -c "import nltk; nltk.download('wordnet', download_dir='/usr/share/nltk_data')"
python -c "import nltk; nltk.download('omw-1.4', download_dir='/usr/share/nltk_data')"
python -c "import nltk; nltk.download('stopwords', download_dir='/usr/share/nltk_data')"
